function removeDuplicate(){
    var temp1 = null;
    var temp2 = null;
    var dup = null; 

    temp1 = this.head;  

    while( temp1 != null && temp1.next != null){
        temp2 = temp1;

        while( temp2.next != null){
            if(temp1.data == temp2.next.data){
                dup = temp2.next;
                temp2.next = temp2.next.next;
            } else {
                temp2 = temp2.next;
            }
        }
        temp1 = temp1.next;
    }
}

    SET next to current  

END WHILE 

After you hit the null, SET prev to node 
return node

END FUNCTION
